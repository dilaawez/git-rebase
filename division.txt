Python
	# Normal division (always returns float)
	print(7 / 2)    # 3.5

	# Floor division (integer division, rounds down)
	print(7 // 2)   # 3

	# Modulus (remainder)
	print(7 % 2)    # 1

Java
	public class DivisionExample {
		public static void main(String[] args) {
			int a = 7, b = 2;

			// Integer division (fractional part is discarded)
			int intResult = a / b;
			System.out.println(intResult); // 3

			// Cast to double for floating-point division
			double doubleResult = (double) a / b;
			System.out.println(doubleResult); // 3.5
	}
}

C
	#include <stdio.h>

	int main() {
		int a = 7, b = 2;

		// Integer division
		int intResult = a / b;
		printf("%d\n", intResult); // 3

		// Floating-point division
		double doubleResult = (double)a / b;
		printf("%f\n", doubleResult); // 3.500000

		 return 0;
}
