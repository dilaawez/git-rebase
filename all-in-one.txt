	# Python example: (a + b) - (c * d) / e
	a = 10
	b = 5
	c = 8
	d = 3
	e = 4

	result = (a + b) - (c * d) / e
	print("Result (float division):", result)

	# If you want integer (floor) division:
	result_int = (a + b) - (c * d) // e
	print("Result (integer // division):", result_int)

java
	public class MathOps {
		public static void main(String[] args) {
			int a = 10, b = 5, c = 8, d = 3, e = 4;

			// Use double cast to ensure floating-point division
			double result = (a + b) - ((double)(c * d) / e);
			System.out.println("Result (double division): " + result);

			// If you keep integers, division truncates:
			int resultInt = (a + b) - (c * d / e); // integer division
		System.out.println("Result (integer division): " + resultInt);
		}
	}

C
	#include <stdio.h>

	int main(void) {
		int a = 10, b = 5, c = 8, d = 3, e = 4;

		// Force floating-point division by casting
		double result = (a + b) - ((double)(c * d) / e);
		printf("Result (double division): %.2f\n", result);

		// Integer arithmetic (truncating division)
		int resultInt = (a + b) - (c * d / e);
		printf("Result (integer division): %d\n", resultInt);

		return 0;
	}
